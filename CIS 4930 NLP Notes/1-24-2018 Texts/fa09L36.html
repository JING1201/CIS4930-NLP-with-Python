class Shape {
  protected Color color;

  public Shape() {
    color = new Color(0, 0, 255);
  }
  public Shape(Color color) {
    this.color = color;
  }

  // public double area() ???
}

class Circle extends Shape {
  private double radius;

  public Circle() {
    super();
    radius = 3.0;
  }
  public Circle(double radius) {
    super();
    this.radius = radius;
  }
  public Circle(double radius, Color color) {
    super(color);
    this.radius = radius;
  }
}
